def least_sq_reg(beta_ls, X, Y):
    beta_ls=np.matrix(beta_ls)
    z=Y-np.dot(X,beta_ls.T)
    return np.dot(z.T,z)

def least_sq_reg_der(beta_ls,X,Y):
    beta_ls=np.matrix(beta_ls)
    pp=-2*np.dot((Y-np.dot(X,(beta_ls).T)).T,X)
    aa= np.squeeze(np.asarray(pp))
    return aa

def least_sq_reg_hess(beta_ls,X,Y):
    ss=2*np.dot(np.transpose(X),X)
    return ss

def quasi_newton(X = X, Y = Y, n_iter = 10000, epsilon = 1e-3, method = 'bfgs'):
    
    (a,b)=X.shape
    beta_lsqn=np.zeros(b) #initial value for beta
    OF_iter=np.zeros(n_iter)
    tol_iter=np.zeros(n_iter)
    i=0
    tol=1000
    b = least_sq_reg_hess(beta_lsqn, X, Y)

    time_start = time.clock()
    while (i <= n_iter-2) and (tol>epsilon):
        i=i+1
        grad=least_sq_reg_der(beta_lsqn,X,Y)
        ddirect=-np.dot(np.linalg.inv(b),grad)#descent direction
        ######Armijo rule to adjust alpha########
        sigma = 0.1
        beta= 0.1
        alpha=1
        while (least_sq_reg(beta_lsqn+alpha*ddirect,X,Y) > least_sq_reg(beta_lsqn,X,Y)+alpha*sigma*np.dot(ddirect,grad)):
            alpha=alpha*beta
        #########################################
        s = beta_lsqn - (beta_lsqn+alpha*ddirect)
        y = least_sq_reg_der(beta_lsqn+alpha*ddirect, X, Y) - least_sq_reg_der(beta_lsqn, X, Y)
        beta_lsqn=beta_lsqn+alpha*ddirect
        OF_iter[i]=least_sq_reg(beta_lsqn, X, Y)
        #tol=np.absolute((OF_iter[i]-OF_iter[i-1])/OF_iter[i]) #tolerance
        tol=np.linalg.norm(grad,ord=2)
        tol_iter[i]=tol
        ##########################################
        ####### UPDATE RULE FOR Bk ###############
        if method == 'bfgs':
            b = b - (np.dot(np.dot(b, s), np.dot(b, s).T)/np.dot(np.dot(s.T, b), s)) + (np.dot(y, y.T)/np.dot(y.T, s))
        elif method == 'symm':
            b = b + np.dot(y - np.dot(b, s), (y - np.dot(b, s)).T)/np.dot((y - np.dot(b, s)).T, s)

    time_elapsed = (time.clock() - time_start)
    
    dic = {'betas': beta_lsqn, 
          'time': time_elapsed, 
          'iter': i, 
          'tol': tol, 
          'error': np.linalg.norm(np.transpose(beta_ls_exact)-beta_lsqn,ord=2)/np.linalg.norm(beta_lsqn,ord=2)}
    
    return dic

def least_absolute_regression(x, y, n, m):
    
    model = pyo.AbstractModel()
    
    model.n = pyo.Param(within=pyo.NonNegativeIntegers) # N√∫mero de muestras
    
    model.m = pyo.Param(within=pyo.NonNegativeIntegers) # N√∫mero de variables
    
    model.n = n #n√∫mero de observaciones

    model.m = m #n√∫mero de variables

    model.I = pyo.RangeSet(1, model.n)

    model.J = pyo.RangeSet(1, model.m+1)

    model.t = pyo.Var(model.I, domain=NonNegativeReals)

    model.bet = pyo.Var(model.J)
    
    model.X = pyo.Param(model.I,model.J,domain = pyo.Reals, mutable=True) # Matriz de X
    
    model.Y = pyo.Param(model.I,default=1, domain = pyo.Reals, mutable=True) # Vector de Y

    def objective(model):
    
        return sum(model.t[i] for i in model.I)
    model.OBJ = pyo.Objective(rule = objective, sense = minimize)
    
    def constraint_upper(model, i):
    
        return (model.Y[i] - sum(model.X[i, j]*model.bet[j] for j in model.J) <= model.t[i])
    
    model.constr_upp = pyo.Constraint(model.I, rule = constraint_upper)
    
    def constraint_lower(model, i):
        
        return (model.Y[i] - sum(model.X[i, j]*model.bet[j] for j in model.J)   >=  -model.t[i])
    
    model.constr_low = pyo.Constraint(model.I, rule = constraint_lower)
    
    Xl = x.tolist()
    Yl = y.tolist()
    
    instance = model.create_instance()
    
    instance.dual = Suffix(direction = Suffix.IMPORT)
    
    # Implementamos el vector Y
    for i in range(n):
        getattr(instance, 'Y')[i+1] = Yl[i][0]

    # Implementamos el vector X
    for i in range(n):
        for j in range(m+1):
            getattr(instance, 'X')[i+1,j+1] = Xl[i][j]
    
    results = opt.solve(instance)
    
    return results, instance.display()
